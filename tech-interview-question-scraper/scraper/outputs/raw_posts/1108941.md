多谢lz总结，通勤那道题为何不能用PQ呢？应该类似Dijkstra吧，应该会比BFS快。
int[][] dir = {{0,1}, {0,-1}, {1,0}, {-1,0}};
int m,n;
public int travel(char[][] grid, int[] times, int[] costs){
m = grid.length;
n = grid[0].length;
int[] src = null, dest = null;
for(int i=0; i<m; i++){
for(int j=0; j<n; j++){
if(grid[i][j] == 's'){
src = new int[]{i, j};
}else if(grid[i][j] == 'd'){
dest = new int[]{i,j};
}
}
}
PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->(a[2] == b[2]? a[3]-b[3]: a[2]-b[2])); // x, y, time, cost, option
pq.offer(new int[]{src[0], src[1], 0, 0, 0});
int[][] seen = new int[m][n];
seen[src[0]][src[1]] = 1;
while(!pq.isEmpty()){
int[] curr = pq.poll();
int r = curr[0];
int c = curr[1];
int time = curr[2];
int cost = curr[3];
int option = curr[4];
System.out.println("visit : " + r + ":" + c + ", option: " + option + ", time: " + time);
if(r == dest[0] && c == dest[1]){
System.out.println("time: " + time);
return option; //return option
}
for(int d=0; d<dir.length; d++){
int nr = r + dir[d][0];
int nc = c + dir[d][1];
if(nr < 0 || nr >= m || nc < 0 || nc >= n || seen[nr][nc] == 1 || grid[nr][nc] == 'x')
continue;
if(option > 0 && ((grid[nr][nc]-'0') != option) && grid[nr][nc] != 'd'){
continue;
}
seen[nr][nc] = 1;
pq.offer(new int[]{nr, nc, time+times[option==0?grid[nr][nc] - '0':option], cost+costs[option],  option==0?grid[nr][nc] - '0':option});
}
}
return -1;
}
