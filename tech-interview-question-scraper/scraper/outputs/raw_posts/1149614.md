class FileMetadata:
def __init__(self, file_id, filename, size, created_at):
self.file_id = file_id
self.filename = filename
self.size = size
self.created_at = created_at
class Storage:
def __init__(self):
self.files = {}  # key: file_id -> (file_content, FileMetadata)
def save_file(self, file_id, content, metadata):
self.files[file_id] = (content, metadata)
def get_file(self, file_id):
return self.files.get(file_id, None)
import time
import uuid
storage = Storage()
# POST /files
def upload_file(request):
file_content = request["File"]
metadata_raw = request["FileMetadata"]
file_id = str(uuid.uuid4())  # 生成唯一ID
metadata = FileMetadata(
file_id=file_id,
filename=metadata_raw.get("filename", "unknown"),
size=len(file_content),
created_at=time.time()
)
storage.save_file(file_id, file_content, metadata)
return {"status": "ok", "fileId": file_id}
# GET /files/{fileId}
def download_file(file_id):
result = storage.get_file(file_id)
if not result:
return {"status": "error", "message": "file not found"}
content, metadata = result
return {
"File": content,
"FileMetadata": {
"filename": metadata.filename,
"size": metadata.size,
"created_at": metadata.created_at
}
}
