def tokenize(text: str, vocab: dict) -> list:
tokens = []
i = 0
# Precompute max word length for optimization
max_len = max((len(w) for w in vocab if w != "UNK"), default=0)
while i < len(text):
matched = False
# Only check up to max word length instead of entire remaining text
for j in range(min(i + max_len, len(text)), i, -1):
s = text
if s in vocab and s != "UNK":
tokens.append(vocab)
i = j
matched = True
break
if not matched:
tokens.append(vocab["UNK"])
i += 1
return tokens

VO Coding Q1
Web Crawler
Single threading
from urllib.parse import urlparse
from collections import deque
class Solution:
def crawl(self, startUrl: str, htmlParser: 'HtmlParser') -> List:
hostname = urlparse(startUrl).hostname
visited = {startUrl}
queue = deque()
while queue:
cur_url = queue.popleft()
urls = htmlParser.getUrls(cur_url)
for url in urls:
if url in visited:
continue
if urlparse(url).hostname == hostname:
queue.append(url)
visited.add(url)
return list(visited)
Multi threading
from concurrent.futures import ThreadPoolExecutor, as_completed
from urllib.parse import urlparse
class Solution:
def crawl(self, startUrl: str, htmlParser: 'HtmlParser') -> List:
hostname = urlparse(startUrl).hostname
visited = {startUrl}
queue = deque()
with ThreadPoolExecutor() as executor:
futures = {executor.submit(htmlParser.getUrls, startUrl)}
while futures:
for future in as_completed(futures):
futures.discard(future)
urls = future.result()
for url in urls:
if url in visited:
continue
if urlparse(url).hostname == hostname:
futures.add(executor.submit(htmlParser.getUrls, url))
visited.add(url)
return list(visited)
