这题目感觉相当的难啊，自己写了一下妥妥超过1个小时了。
我的思路是参考数据库的的snapshot isolation，用snapshot id或者version id，但不是每次call iterator()都为每个value添加snapshot id。比如一开始初始化snapshot id = 0，然后 add(15), add(8), 然后我即使call 5次 iterator()，每个iterator里会存自己对应的snapshot id，但是15和8的node里仍旧只存snapshotId = 0，因为自snapshot 0 之后，数据再也没发生变化。但如果这个时候remove(15)，那么在15对应的node里要加入在snapshotId = 5的时候被删除了这个信息，我是用一个arraylist来存的，ArrayList里只存snapshot id，存的第奇数个snapshot id就表示在这个snapshot里add了，第偶数个则是remove了。
public class SnapNode<T> {
public T val;
private List<Integer> versions;
public SnapNode<T> next;
public SnapNode(T val, int initalVersion) {
this.val = val;
this.next = null;
this.versions = new ArrayList<>();
this.versions.add(initalVersion);
}
// 0: removed, 1: existing, -1: never-added
public int isExisting(int versionId) {
int size = this.versions.size();
int v = size;
// Can use binary search to optimize
for (int i = 0; i < size; i++) {
if (this.versions.get(i) > versionId) {
v = i;
break;
}
}
if (v <= 0) return -1;
return v & 1;
}
public void add(int versionId) {
var isExisting = this.isExisting(versionId);
if (isExisting != 1) {
// handle: remove before an add
if (handleLastItem(versionId)) return;
this.versions.add(versionId);
}
}
public void remove(int versionId) {
if (this.isExisting(versionId) == 1) {
// handle: add before a remove
if (handleLastItem(versionId)) return;
this.versions.add(versionId);
}
}
private boolean handleLastItem(int versionId) {
int lastIndex = versions.size() - 1;
if (lastIndex >= 0 && this.versions.get(lastIndex) == versionId) {
this.versions.remove(lastIndex);
return true;
}
return false;
}
}

跟 brucewill1 一样的思路，用C++实现了一遍。简单起见，每个add和remove都会increment snapshot version。
struct Node {
int value;
vector<int> snapshots;
};
class Iterator {
public:
Iterator(list<Node>::iterator begin, list<Node>::iterator end, int version)
: it(begin), end(end), version(version) {}
// is_end() == false
int ref() { return it->value; }
// is_end() == false
void advance() { it++; }
bool is_end() {
next_valid();
return it == end;
}
private:
void next_valid() {
while (it != end) {
vector<int>::iterator vi = ranges::upper_bound(it->snapshots, version);
if ((vi - it->snapshots.begin()) % 2 == 1) {
break;
}
it++;
}
return;
}
list<Node>::iterator it;
list<Node>::iterator end;
int version;
};
class SnapshotSet {
public:
SnapshotSet() : version(0) {}
void add(int value) {
if (!m.contains(value)) {
Node node = {value, {}};
nodes.push_back(node);
m[value] = prev(nodes.end());
}
// already added.
if (m[value]->snapshots.size() % 2 == 1) return;
m[value]->snapshots.push_back(version++);
}
void remove(int value) {
if (!m.contains(value)) return;
// already removed.
if (m[value]->snapshots.size() % 2 == 0) return;
m[value]->snapshots.push_back(version++);
}
bool contains(int value) {
if (!m.contains(value)) return false;
return m[value]->snapshots.size() % 2 == 1;
}
Iterator iterator() {
return Iterator(nodes.begin(), nodes.end(), version-1);
}
private:
list<Node> nodes;
unordered_map<int, list<Node>::iterator> m;
int version;
};
