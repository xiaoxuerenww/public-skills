一道题，地理面筋
bfs 求 shortest time/cost from src to dst in 2d grid.
优化是 扫一遍，而不是扫4遍。
附上python解法
from collections import deque
class Solution:
def bfs(self, grid, start, end, mode):
m, n = len(grid), len(grid[0])
queue = deque([(start[0], start[1], 0)])
visited = set([start])
while queue:
x, y, d = queue.popleft()
if (x, y) == end:
return d
for dx, dy in [(1,0), (0,1), (-1, 0), (0,-1)]:
nx, ny = dx + x, dy+y
if 0<= nx < m and 0<= ny < n and (nx, ny) not in visited:
val = grid[nx][ny]
if val == 'X':
continue
if val == str(mode) or val == 'D' or val == 'S':
queue.append((nx, ny, d+1))
visited.add((nx, ny))
return -1
def getMode(self, grid):
modes = [1, 2, 3, 4]
time_map = [3, 2, 1, 1]
cost_map = [0, 1, 3, 2]
best_mode = -1
min_time = float('inf')
min_cost = float('inf')
start_pos = None
dest_pos = None
rows = len(grid)
cols = len(grid[0])
# 1. 定位起点 S 和终点 D
for r in range(rows):
for c in range(cols):
if grid[r][c] == 'S':
start_pos = (r, c)
elif grid[r][c] == 'D':
dest_pos = (r, c)
for m in modes:
steps = self.bfs(grid, start_pos, dest_pos, m)
print(steps)
if steps != -1: # 如果能到达
curr_time = steps * time_map[m-1]
curr_cost = steps * cost_map[m-1]
if curr_time < min_time:
min_time, min_cost, best_mode = curr_time, curr_cost, m
elif curr_time == min_time:
if curr_cost < min_cost:
min_cost, best_mode = curr_cost, m
return best_mode
