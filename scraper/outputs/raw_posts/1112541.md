Yes, there should be a significant performance difference! Mayne you're using asyncio and block (await) every call which makes it of not value. Make sure you're really parallelizing execution. You can use an LLM to help you improve your code
Here's sample code:
import asyncio
class Solution:
def crawl(self, startUrl: str, htmlParser: 'HtmlParser') -> set:
return asyncio.run(self.async_crawl(startUrl, htmlParser))
async def async_crawl(self, startUrl: str, htmlParser: 'HtmlParser') -> set:
visited = set()
queue = asyncio.Queue()
host = startUrl.split('/')
async def worker():
while True:
url = await queue.get()
if url in visited:
queue.task_done()
continue
visited.add(url)
# async with semaphore:
new_urls = await asyncio.to_thread(htmlParser.getUrls, url)
for u in new_urls:
if u not in visited and u.split('/') == host:
queue.put_nowait(u)
queue.task_done()
queue.put_nowait(startUrl)
workers = [asyncio.create_task(worker()) for _ in range(2)]
await queue.join()
for w in workers:
w.cancel()
return visited
